count <- 0
for (i in 1:nrow(final_q6s)) {
  days <- 0
  curr_id <- final_q6s[i, "id"]
  for (j in 1:final_q6s[i, "max(q6)"]$'max(q6)') {
    full_df[count,] <- c(curr_id, j, days)
    days <- days + 100
    count <- count + 1
  }
}
